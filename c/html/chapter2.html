例2.2 判定一个年份是否是闰年

[解题思路]：设要判断的年份存在变量year中，其是闰年的判断依据是

  1. 能被4整除，但不能被100整除的是闰年，如2008， 1996
  2. 或者，能被400整除的是闰年， 如2000
  3. 其他情况下都是非闰年， 如2100
  4. 判断能被400整除必然能被4整除，所以先判断400，再判断4

例2.3 求n!(阶乘）

[解题思路]：设n是大于1的整数，阶乘的值保存在变量factorial中，可以用公式： factorial=1*2*3*...*(n-1)*n，这样我们可以用一个变量i来循环取1~n的值，分别把其值和factorial相乘，即可得到所需要的值。

例2.6 给定一个年份判断是否是闰年

#include<stdio.h>
#define FALSE 0
#define TRUE  !FALSE

int main(void)
{
      int year, is_leap_year = FALSE;
      printf("input year:");
      scanf("%d", &year);
      
      if(year % 400 == 0)
            is_leap_year = TRUE;
      else
            if( year % 4 == 0)
                  if( year % 100 == 0)
                        is_leap_year = FALSE ;
                  else
                        is_leap_year = TRUE ;
            else
                  is_leap_year = FALSE;
      
      if(is_leap_year == TRUE)
            printf("%d is leap year.\n", year);
      else
            printf("%d is NOT leap year.\n", year);
      
      return 0;
}

例2.7 判断一个整数是否是素数（质数）

#include<stdio.h>
#include<math.h>
#define FALSE 0
#define TRUE  !FALSE

int main(void)
{
      int n, is_prime = TRUE, i;
      printf("input number:");
      scanf("%d", &n);
      
      i = 2;
      while(i<sqrt(n) && is_prime == TRUE ){
            if( n % i == 0)
                  is_prime = FALSE;
            else
                  i = i + 1;            
      }

      if( is_prime == TRUE )
            printf("%d is prime number.\n", n);
      else
            printf("%d is NOT prime number.\n", n);
      
      return 0;
}

2.3 算法的特性

  1. 有穷性：必须是有限步骤的；
  2. 确定性：每一个步骤都是确定的，不能含糊或模棱两可
  3. 有0个或多个输入
  4. 至少有1个输出
  5. 有效性：能有效被执行，得到确定的结果。

4.7 求两个整数的最大公约数，N-S流程图和C语言程序

#include<stdio.h>

int main()
{
    int m,n,i;
    printf("input 2 integer number:");
    scanf("%d,%d", &m, &n);

    if(m==n)
    {
        printf("They are the same numbers.\n");
        return -1;
    }

    if(m>n)
    {
        int t = m;   t = n;   n = m;      m=t;
    }

    i=m;
    while(i>0)
    {
        if(m % i == 0 && n % i == 0)
        {
            printf("The greatest common divisor of %d and %d is %d", 
               m, n, i);
            return 0;
        }
        i = i - 1;
    }

    return -1;
}

4.7 求\( ax^2+bx+c=0 \) 的解，分别考虑
【1】有两个不等实根
【2】有两个相等实根的情况。
写出N-S流程图和C语言程序

#include<stdio.h>
#include <math.h>
#define VERY_SMALL 1e-9

int main()
{
    double a,b,c, d;

    printf("input a,b,c:");
    scanf("%lf,%lf,%lf", &a, &b, &c);

    d = b*b - 4*a*c;

    if(fabs(d)<VERY_SMALL)
    {
        printf("There are 2 same roots of this equation: %f\n", -b/(2*a));
        return 0;
    }
    else
    {
        if(d<0)
        {
            printf("There is no REAL root of this equation.\n");
            return 0;
        }
        else
        {
            double x1,x2;
            x1 = (-b+sqrt(d))/(2*a);
            x2 = (-b-sqrt(d))/(2*a);
            printf("The real roots of this equation is:\n");
            printf("x1 = %f\n", x1);
            printf("x2 = %f\n", x2);

        }
    }
    return 0;
}

4.6. 用结构化设计程序，将100-200之间的素数输出，写出N-S程图和C语言程序

#include<stdio.h>
#include <math.h>

#define FALSE 0
#define TRUE  !FALSE

int prime(int n);

int main()
{
    int m = 100;
    while(m<=200)
    {
        if(prime(m) == TRUE)
            printf("%d is prime.\n", m);
        m = m +1;
    }
    return 0;
}

int prime(int n)
{
    int sqrt_n = (int) sqrt(n);
    int i, is_prime = TRUE;

    i=2;
    while(i<=sqrt_n && is_prime==TRUE)
    {
        if( n % i == 0)
            is_prime = FALSE;
        else
            i = i +1;
    }
    return is_prime;
}
